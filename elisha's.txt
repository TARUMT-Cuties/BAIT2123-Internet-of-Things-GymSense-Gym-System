#include <WiFi.h>
#include <HTTPClient.h>
#include <Wire.h>
#include <Adafruit_GFX.h>
#include <Adafruit_SSD1306.h>

const char* ssid = "applejuice";
const char* password = "12345678";

#define SCREEN_WIDTH 128
#define SCREEN_HEIGHT 64
Adafruit_SSD1306 display(SCREEN_WIDTH, SCREEN_HEIGHT, &Wire, -1);

// Pins
#define trigPin 5
#define echoPin 18
#define buzzer 19

// MPU
const int MPU = 0x68;
int16_t AcY;

// workout
int reps = 0, target = 12, sets = 1, currentSet = 1;
bool running = false;
String exercise = "squat";

bool squatDown  = false;
bool curlDown   = false;
bool pushupDown = false;

bool placement = false;

// timing
unsigned long lastDisplay = 0;
unsigned long lastControl = 0;
unsigned long lastRepTime = 0;
const unsigned long MIN_REP_INTERVAL = 1000; // ms — ignore reps faster than this

// ================= BUZZER =================
void beepRep() {
  digitalWrite(buzzer, HIGH);
  delay(200);
  digitalWrite(buzzer, LOW);
}

void beepSetDone() {
  for (int i = 0; i < 2; i++) {
    digitalWrite(buzzer, HIGH);
    delay(300);
    digitalWrite(buzzer, LOW);
    delay(150);
  }
}

void beepWorkoutDone() {
  for (int i = 0; i < 3; i++) {
    digitalWrite(buzzer, HIGH);
    delay(400);
    digitalWrite(buzzer, LOW);
    delay(150);
  }
}

// ================= SEND =================
void sendData(int repCount, int setNum, bool done) {
  if (WiFi.status() != WL_CONNECTED) return;

  HTTPClient http;
  http.begin("http://172.20.10.3:3000/workout");
  http.addHeader("Content-Type", "application/json");

  String json = "{";
  json += "\"exercise\":\"" + exercise + "\",";
  json += "\"reps\":" + String(repCount) + ",";
  json += "\"set\":" + String(setNum) + ",";
  json += "\"done\":" + String(done ? "true" : "false");
  json += "}";

  http.POST(json);
  http.end();
}

// ================= CONTROL =================
void getControl() {
  if (millis() - lastControl < 1000) return;
  lastControl = millis();

  if (WiFi.status() != WL_CONNECTED) return;

  HTTPClient http;
  http.begin("http://172.20.10.3:3000/control");

  int code = http.GET();

  if (code == 200) {
    String payload = http.getString();

    bool newRunning = payload.indexOf("\"running\":true") >= 0;

    String newExercise = "";
    if (payload.indexOf("\"exercise\":\"curl\"") >= 0)   newExercise = "curl";
    if (payload.indexOf("\"exercise\":\"squat\"") >= 0)  newExercise = "squat";
    if (payload.indexOf("\"exercise\":\"pushup\"") >= 0) newExercise = "pushup";

    if ((newRunning && !running) || newExercise != exercise) {
      reps = 0;
      currentSet = 1;
      squatDown  = false;
      curlDown   = false;
      pushupDown = false;
    }

    placement  = payload.indexOf("\"placement\":true") >= 0;
    running    = newRunning;
    if (newExercise != "") exercise = newExercise;

    int tIndex = payload.indexOf("target");
    if (tIndex >= 0) target = payload.substring(tIndex + 8).toInt();

    int sIndex = payload.indexOf("sets");
    if (sIndex >= 0) sets = payload.substring(sIndex + 6).toInt();
  }

  http.end();
}

// ================= DISPLAY =================
void updateDisplay() {
  if (millis() - lastDisplay < 200) return;
  lastDisplay = millis();

  display.clearDisplay();
  display.setTextSize(1);
  display.setTextColor(WHITE);

  if (placement) {
    // ── Placement screen ──────────────────────────────────
    display.setCursor(0, 0);
    display.println("PLACE SENSOR");

    display.drawFastHLine(0, 10, 128, WHITE);

    display.setCursor(0, 14);
    display.print("Exercise: ");
    display.println(exercise);

    display.setCursor(0, 26);
    if (exercise == "squat") {
      display.println("Floor, facing");
      display.setCursor(0, 35);
      display.println("up at ur legs");
    } else if (exercise == "pushup") {
      display.println("Under chest,");
      display.setCursor(0, 35);
      display.println("facing up");
    } else if (exercise == "curl") {
      display.println("Strap MPU to");
      display.setCursor(0, 35);
      display.println("your forearm");
    }

    display.setCursor(0, 50);
    display.println("Press READY on app");
  } else {
    // ── Normal workout screen ─────────────────────────────
    display.setCursor(0, 0);
    display.println("SMART GYM");

    display.setCursor(0, 12);
    display.print("Exercise: ");
    display.println(exercise);

    display.setCursor(0, 24);
    display.print("Set: ");
    display.print(currentSet);
    display.print("/");
    display.println(sets);

    display.setCursor(0, 36);
    display.print("Reps: ");
    display.print(reps);
    display.print("/");
    display.println(target);

    display.setCursor(0, 52);
    display.println(running ? "RUNNING" : "WAITING");
  }

  display.display();
}

// ================= SETUP =================
void setup() {
  Serial.begin(115200);

  WiFi.begin(ssid, password);
  while (WiFi.status() != WL_CONNECTED) {
    delay(500);
  }

  pinMode(trigPin, OUTPUT);
  pinMode(echoPin, INPUT);
  pinMode(buzzer, OUTPUT);

  // 🔥 TEST BUZZER (IMPORTANT)
  digitalWrite(buzzer, HIGH);
  delay(500);
  digitalWrite(buzzer, LOW);

  Wire.begin(21,22);
  Wire.setClock(100000);
  delay(100);

  display.begin(SSD1306_SWITCHCAPVCC, 0x3C);

  display.clearDisplay();
  display.setTextSize(2);
  display.setCursor(20,20);
  display.println("READY");
  display.display();
  delay(1000);

  // MPU wake
  Wire.beginTransmission(MPU);
  Wire.write(0x6B);
  Wire.write(0);
  Wire.endTransmission(true);
}

// ================= LOOP =================
void loop() {

  getControl();
  updateDisplay();

  if (!running) return;

  // ===== SQUAT =====
  if (exercise == "squat") {

    digitalWrite(trigPin, LOW);
    delayMicroseconds(2);
    digitalWrite(trigPin, HIGH);
    delayMicroseconds(10);
    digitalWrite(trigPin, LOW);

    long duration = pulseIn(echoPin, HIGH, 30000);
    float distance = duration * 0.034 / 2;

    if (distance > 0 && distance < 15) squatDown = true;

    if (distance > 25 && squatDown && millis() - lastRepTime >= MIN_REP_INTERVAL) {
      reps++;
      squatDown = false;
      lastRepTime = millis();

      beepRep();
      sendData(reps, currentSet, false);
    }
  }

  // ===== PUSHUP =====
  else if (exercise == "pushup") {

    digitalWrite(trigPin, LOW);
    delayMicroseconds(2);
    digitalWrite(trigPin, HIGH);
    delayMicroseconds(10);
    digitalWrite(trigPin, LOW);

    long duration = pulseIn(echoPin, HIGH, 30000);
    float distance = duration * 0.034 / 2;

    // Sensor faces down from chest: close to floor = down, far = up
    if (distance > 0 && distance < 10) pushupDown = true;

    if (distance > 20 && pushupDown && millis() - lastRepTime >= MIN_REP_INTERVAL) {
      reps++;
      pushupDown = false;
      lastRepTime = millis();

      beepRep();
      sendData(reps, currentSet, false);
    }
  }

  // ===== CURL =====
  else {

    Wire.beginTransmission(MPU);
    Wire.write(0x3B);
    Wire.endTransmission(false);
    Wire.requestFrom(MPU, 6, true);

    if (Wire.available() == 6) {
      Wire.read(); Wire.read();
      AcY = Wire.read() << 8 | Wire.read();
    }

    if (AcY < -12000) curlDown = true;

    if (AcY > 12000 && curlDown && millis() - lastRepTime >= MIN_REP_INTERVAL) {
      reps++;
      curlDown = false;
      lastRepTime = millis();

      beepRep();
      sendData(reps, currentSet, false);
    }
  }

  // ===== SET DONE =====
  if (reps >= target) {

    sendData(target, currentSet, true);

    currentSet++;
    reps = 0;
    squatDown  = false;
    curlDown   = false;
    pushupDown = false;

    if (currentSet > sets) {
      running = false;
      currentSet = 1;
      beepWorkoutDone();
    } else {
      beepSetDone();
    }
  }
}
